%-------------------------------------------------------------------------------
\subsubsection{Polynôme caractéristique d'une matrice $2 \times 2$}
%-------------------------------------------------------------------------------

On considère la matrice 
  $$
  A = \left[\begin{array}{cc} a & b \\ c & d\end{array}\right]
  $$
  
  \begin{enumerate}
% [Ex 1.3.3]
  \item Déterminer son polynôme caractéristique et donner les conditions sur $p = \tr(A)$ et $q = \det(A)$ pour que $A$ admette deux valeurs propres réelles.
  \solution{On a 
  \begin{align*}
    P_A(\lambda) 
    & = |A - \lambda I_2| 
    = \left|\begin{array}{cc} a - \lambda & b \\ c & d - \lambda \end{array}\right|
    = (a - \lambda)( d - \lambda) - bc \\
    & = \lambda^2 - (a+d) \lambda + (ad - bc) 
    = \lambda^2 - p \lambda + q
  \end{align*}
  }
  %
  \item Donner les conditions sur $p = \tr(A)$ et $q = \det(A)$ pour que $A$ admette deux valeurs propres réelles.
  \solution{Le discrimant de $P_A$ est $\Delta = p^2 - 4 q$. 
  $P_A$ admet donc deux solutions réelles (éventuellement égales) ssi
  $$
  p^2 - 4 q \geq 0
  \qquad \Leftrightarrow \qquad
  q \leq p^2 / 4.
  $$}
\end{enumerate}
