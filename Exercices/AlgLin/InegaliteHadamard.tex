Soit $A$ une matrice de $\Mcal_n$ inversible dont on note les vecteurs colonnes $u_1, \dots u_n$ : 
$$
A = [u_1 \dots u_n].
$$
L'objectif est de montrer l'inégalité d'Hadamard qui affirme que
$$
|A| \leq \prod_{i=1}^n \|v_i\|.
$$

\paragraph{Préliminaire.}
$$
\left(\prod_{i=1}^n a_i\right)^{1/n} \leq \frac1n \sum_{i=1}^n a_i
$$
\todo{Preuve par Jensen:}
$$
\log\left(\frac1n \sum_{i=1}^n a_i\right)
\geq \frac1n \sum_{i=1}^n \log(a_i)
= \frac1n \log\left(\prod_{i=1}^n a_i\right)
= \left(\prod_{i=1}^n a_i\right)^{1/n}
$$
